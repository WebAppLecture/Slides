var baseList = $('article > section');
console.log('baseList', baseList);//Loggen

//Die Basisauswahl filtern (1)
console.log('filter1', filter1);//Loggen
//Die Basisauswahl filtern (2)
var filter2 = baseList.eq(7);
console.log('filter2', filter2);//Loggen
//Die Basisauswahl filern (3)
var filter3 = baseList.filter('#lec08-example1');
console.log('filter3', filter3);//Loggen
//Die gefilterte (1) Basisauswahl filtern (3)
