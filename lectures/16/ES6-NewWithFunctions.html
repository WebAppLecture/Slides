    function tellMeMore([a,b,...rest])
    {
      console.log("a: "+a+ " | b: "+b);
      if(rest.length>0)
        for(let i = 0; i < rest.length;i++)
          console.log(rest[i]);
    }
    let a = [1,2];
    let b = [1,2,3,4,5];
    tellMeMore(a); // "a: 1 | b: 2"
    tellMeMore(b); // "a: 1 | b: 2 " + alle anderen Elemente in jeweils einer neuen Zeile
