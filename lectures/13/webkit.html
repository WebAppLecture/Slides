namespace WebCore { // Namensraum für Webkit-HTML Objekte
	class CanvasPattern;
	class HTMLCanvasElement;
	/* Mehr class Prototypen */
	class CanvasRenderingContext { //Oberklasse für Context-Elemente
		WTF_MAKE_NONCOPYABLE(CanvasRenderingContext);
		WTF_MAKE_FAST_ALLOCATED;//Ausnutzen von Macros
	public: //Öffentlich zugängliche Methoden
		virtual ~CanvasRenderingContext() { }
		void ref() { m_canvas->ref(); }
		void deref() { m_canvas->deref(); }
		HTMLCanvasElement* canvas() const { return m_canvas; }
		virtual bool is2d() const { return false; }
		virtual bool is3d() const { return false; }
		virtual bool isAccelerated() const { return false; }
		virtual void paintRenderingResultsToCanvas() {}
		virtual bool paintsIntoCanvasBuffer() const { return true; }
	protected: //Nur für die Vererbung bestimmt
		CanvasRenderingContext(HTMLCanvasElement*); /* ... */
	private: //Nur für klasseninterne Verwendung bestimmt
		HTMLCanvasElement* m_canvas; //Referenz auf zugehöriges Canvas
		HashSet m_cleanURLs;
	};
}
