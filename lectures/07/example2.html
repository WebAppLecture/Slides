$(document).ready(function() {//Verwenden die jQuery Ready Methode
	//Erstellt einen Datumsstring im Format YYYY-MM-DD mit dem aktuellen Datum
	document.getElementById('ex15-date').value = new Date().toISOString().substr(0,10);
	if(document.getElementById('ex15-range').type == 'text') //Fallback notwendig?!
		$('<div id="ex15-range"></div>').replaceAll('#ex15-range').slider({
			slide: function(event, ui) { ex15shownumber(ui.value); }//Slider-Change Event
		});//Beim Slider benötigen wir leider ein div - daher $().replaceAll('') notwendig
	if(document.getElementById('ex15-date').type == 'text')	//Beim Datepicker sieht das anders aus
		$('#ex15-date').datepicker({ dateFormat: 'yy-mm-dd' });
});//Methode um den aktuellen Wert des Sliders anzuzeigen
function ex15shownumber(value) { $('#ex15-shownumber').text(value); }
